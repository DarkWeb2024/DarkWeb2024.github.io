Overview Module-1 P-2 Histogram to check frequency distribution 
import pandas as pd
import matplotlib.pyplot as plt

mtcars = pd.read_csv(r'C:\Users\HACKME\Downloads\mtcars.csv')

plt.hist(mtcars['mpg'], bins=10, color='skyblue', edgecolor='black')

plt.xlabel('Miles per gallon (mpg)')
plt.ylabel('Frequency')
plt.title('Histogram of Miles per gallon (mpg)')

plt.show()
